4TH VIDEO   CREATE DB AND TABLE
-------------------------------
CREATE DATABASE OF NOT EXISTS school;

USE school;

CREATE TABLE IF NOT EXISTS students(
id INT AUTO_INCREMENT PRIMARY_KEY,
first_name VARCHAR(50),
last_name VARCHAR(50),
);
