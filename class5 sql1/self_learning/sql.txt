between operator 
select * from payments where payment_value between 150 and 200;
(will show all data where payment_value  is from 150 to 200 we can also do this by >= and <= but this is short query)

in operator
select * from payments where payment_type in ("MCB","HBL",NBP,UBL);
(SHOW ALL DATA FROM MCB, HBL, NBP, UBL)
 
not in operator
select * from payments where payment_type not in ("MCB","HBL",NBP,UBL);
(show all data excluding MBB,HBL.NBP,UBL)


like operator
select customer_city from customers  where customer_city like "r%";
(show all city from customers which start from  r)

select customer_city from customers  where customer_city like "%r";
(show all city from customers which end from  r)

select customer_city from customers  where customer_city like "%de";
(show all city from customers which have de in name either in mid , end , start)

orderby
select * from payments order by payment_value;
(will show all data from payments table assending by payment_value col 0,1,2,3)

order by  dec
select * from payments order by payment_value dec;
(will show all data from payments table decending by payment_value col 10,9,9,8..)


select * from payments order by payment_value, payment_type desc;
(will show payment_value like 0,1,2...9, and payment_type like 10,9,..)

select * from payments where payment_installment = 1 order by payment_value; 
(show all data from payment table when payme is 1 and show all in assending order)

select * from payments where payment_installment = 1 order by payment_value dec;

Limit
select * from payments limit 5;
(will show any 5 rows)

select * from payments limit 2,3;
(show shop top 3 row after excluding top 2 means  3,4,5th row)
